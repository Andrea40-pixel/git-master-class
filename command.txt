git config --list (cmd for listing conf detail)
git config --global user.name "username"(cmd for adding username)
git config --global user.email "abc@gmail.com"
git init (cmd for intializing git repository)
git status (for untracked files)
git add -A (cmd for adding all untracked files to staging area)
git commit -m "commit mssg" (comd for adding all staged files to local repository)
git remote add name remote_path ( command for adding remote repository)
git remote -v( command for lifting remote)
git push -u remote_name branch_name (command for adding changes in lr to rr)
git branch(listing all branch)
git log(commit history)
